Le code ci-dessous n'utilise pas les it√©rateurs et, par cons√©quent, doit stocker en

```python
import math

def main():
	# √âtape 1 : mat√©rialiser toute la plage (grosse liste inutile)
	numbers = list(range(50_000_000))

	# √âtape 2 : filtrer les multiples de 7 (liste interm√©diaire)
	multiples_de_7 = [n for n in numbers if n % 7 == 0]

	# √âtape 3 : s√©lectionner les carr√©s parfaits dont la racine est paire (liste interm√©diaire)
	carres_sqrt_pair = []

		if r * r == n and r % 2 == 0:
			carres_sqrt_pair.append(n)

	# √âtape 4 : transformer (liste interm√©diaire)
	transformes = [(n // 7) ** 2 for n in carres_sqrt_pair]

	# √âtape 5 : ignorer les 100 premiers
	apres_saut = transformes[100:]

	# √âtape 6 : prendre les 100 suivants
	selection = apres_saut[:100]

	# √âtape 7 : agr√©gations
	total = sum(selection)
	mini = min(selection) if selection else None
	maxi = max(selection) if selection else None
	moyenne = total / len(selection) if selection else None

	print(f"Somme : {total}")
	print(f"Min/Max : {mini}/{maxi}")
	print(f"Moyenne : {moyenne}")

if __name__ == "__main__":
	main()
```
